#include <Wire.h>
#include "RTClib.h"
#include "SoftwareSerial.h"
#include "DFRobotDFPlayerMini.h"

SoftwareSerial mySoftwareSerial(10, 11); 
DFRobotDFPlayerMini myDFPlayer;
RTC_DS3231 rtc;

// ================= PINS =================
const int ignitionPin   = 2;
const int leftSignalPin = 3;
const int rightSignalPin = 4;
const int doorPin       = 5;

// ================= STATES =================
bool isEngineOn = false;
bool leftActive = false;
bool rightActive = false;
bool doorState = false;

// ================= TIMERS =================
unsigned long lastLeftTime = 0;
unsigned long lastRightTime = 0;
unsigned long doorWaitStartTime = 0; // සවුන්ඩ් එක ප්ලේ වෙලා ඉවර වුණු වෙලාව
bool isWaitingForNextDoorPlay = false; // විවේක කාලයේ සිටිනවාදැයි බැලීමට

const int signalTimeout = 700;
const int doorLoopInterval = 5000; // විවේක කාලය (තත්පර 5)

// ================= SOUND CONTROL =================
bool isPlaying = false;
unsigned long playStartTime = 0;
int playDuration = 2000;

void playSound(int track, int durationMs, bool loopMode=false) {
  myDFPlayer.stop();
  delay(100);
  if(loopMode) myDFPlayer.loop(track);
  else myDFPlayer.play(track);

  isPlaying = true;
  playStartTime = millis();
  playDuration = durationMs;
}

// ================= SETUP =================
void setup() {
  Serial.begin(9600);
  mySoftwareSerial.begin(9600);

  pinMode(ignitionPin, INPUT);
  pinMode(leftSignalPin, INPUT);
  pinMode(rightSignalPin, INPUT);
  pinMode(doorPin, INPUT);

  if (!rtc.begin()) {
    Serial.println("RTC Not Found!");
    while (1);
  }

  if (!myDFPlayer.begin(mySoftwareSerial)) {
    Serial.println("DFPlayer Error!");
    while(true);
  }

  myDFPlayer.volume(30);
  Serial.println("System Ready...");
}

// ================= LOOP =================
void loop() {
  unsigned long now = millis();

  // සවුන්ඩ් එකක් ප්ලේ වෙන වෙලාව අවසන් දැයි බැලීම
  if (isPlaying && (now - playStartTime > playDuration)) {
    isPlaying = false;
    
    // දොරේ සවුන්ඩ් එක ප්ලේ වෙලා ඉවර වුණු ගමන් විවේක කාලය මැනීම ආරම්භ කරන්න
    if (doorState) { 
      doorWaitStartTime = millis(); 
      isWaitingForNextDoorPlay = true; 
    }
  }

  // ================= IGNITION =================
  int ignition = digitalRead(ignitionPin);

  if (ignition == HIGH && !isEngineOn) {
    DateTime timeNow = rtc.now();
    int hour = timeNow.hour();
    
    if (hour >= 0 && hour < 12) {
      playSound(1, 4000); 
      delay(4000);        
    } 
    else if (hour >= 12 && hour < 16) {
      playSound(2, 5000); 
      delay(5000);        
    } 
    else {
      playSound(3, 5000); 
      delay(5000);        
    }

    playSound(7, 7000);   
    isEngineOn = true;
  }

  if (ignition == LOW && isEngineOn) {
    playSound(8, 3000); 
    isEngineOn = false;
  }

  // ================= LEFT/RIGHT SIGNALS (කලින් කෝඩ් එකමයි) =================
  if (digitalRead(leftSignalPin) == HIGH) {
    lastLeftTime = now;
    if (!leftActive && !isPlaying) { playSound(4, 0, true); leftActive = true; }
  }
  if (leftActive && (now - lastLeftTime > signalTimeout)) { myDFPlayer.stop(); leftActive = false; }

  if (digitalRead(rightSignalPin) == HIGH) {
    lastRightTime = now;
    if (!rightActive && !isPlaying) { playSound(5, 0, true); rightActive = true; }
  }
  if (rightActive && (now - lastRightTime > signalTimeout)) { myDFPlayer.stop(); rightActive = false; }

  // ================= DOOR LOOP LOGIC (FIXED) =================
  int doorNow = digitalRead(doorPin);

  if (doorNow == HIGH) {
    if (!doorState) { // දොර දැන්ම ඇරියා නම් විවේකයකින් තොරව ප්ලේ කරන්න
      doorState = true;
      isWaitingForNextDoorPlay = false; 
    }

    // සවුන්ඩ් එකක් ප්ලේ වෙන්නේ නැත්නම් සහ විවේක කාලය (5s) ඉවර නම් ප්ලේ කරන්න
    if (!isPlaying) {
      if (!isWaitingForNextDoorPlay || (now - doorWaitStartTime > doorLoopInterval)) {
        Serial.println("Door Open - Playing Sound");
        playSound(6, 5000); // Track 6 ප්ලේ කරන්න (තත්පර 2ක් ඇතුළත)
        isWaitingForNextDoorPlay = false; 
      }
    }
  } else {
    doorState = false;
    isWaitingForNextDoorPlay = false;
  }
}
